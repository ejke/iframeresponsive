/*This solution might help in case if video needs to take max 1/4 of screen space when fullscreen. In that case scrollbar becomes visible on 796px width and this solves it, making video fullwidth on smaller screensizes. */ /*html*/
/*css*/ @media screen and (max-width: 796px) { .video-container { position: relative; padding-bottom: 51%; padding-top: 30px; height: 0; overflow: hidden; } .video-container iframe { position: absolute; top:0; left: 0; width: 100%; height: 100%; } } /*And to be pedant, you could add */ @media screen and (max-width: 610px) { .video-container { padding-bottom: 48%; } } @media screen and (max-width: 415px) { .video-container { padding-bottom: 45%; } } @media screen and (max-width: 313px) { .video-container { padding-bottom: 42%; } }
